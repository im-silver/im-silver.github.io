sealed class Either<out L, out R> {
    data class Left<L>(val value: L) : Either<L, Nothing>()
    data class Right<R>(val value: R) : Either<Nothing, R>()

    /** 성공(R) 값 변환 */
    fun <T> map(transform: (R) -> T): Either<L, T> =
        when (this) {
            is Left -> this
            is Right -> Right(transform(value))
        }

    /** 실패(L) 값 변환 */
    fun <T> mapLeft(transform: (L) -> T): Either<T, R> =
        when (this) {
            is Left -> Left(transform(value))
            is Right -> this
        }

    /** R 값에 대한 체이닝(flatMap) */
    fun <T> flatMap(transform: (R) -> Either<L, T>): Either<L, T> =
        when (this) {
            is Left -> this
            is Right -> transform(value)
        }

    /** L 값에 대한 체이닝(flatMap) */
    fun <T> flatMapLeft(transform: (L) -> Either<T, R>): Either<T, R> =
        when (this) {
            is Left -> transform(value)
            is Right -> this
        }

    /** Left, Right를 모두 처리해서 하나의 값으로 변환 */
    fun <T> fold(onLeft: (L) -> T, onRight: (R) -> T): T =
        when (this) {
            is Left -> onLeft(value)
            is Right -> onRight(value)
        }

    /** 성공 값 가져오기 (없으면 null) */
    fun getOrNull(): R? =
        when (this) {
            is Left -> null
            is Right -> value
        }

    /** 실패 값 가져오기 (없으면 null) */
    fun getLeftOrNull(): L? =
        when (this) {
            is Left -> value
            is Right -> null
        }
}
